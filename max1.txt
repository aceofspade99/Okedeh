function isMobileDevice() {
    if (empty($_SERVER['HTTP_USER_AGENT'])) {
        return false;
    }
    $ua = strtolower($_SERVER['HTTP_USER_AGENT']);
    
    $mobileKeywords = [
        'android', 'iphone', 'ipad', 'ipod', 'mobile', 'windows phone',
        'webos', 'blackberry', 'opera mini', 'opera mobi', 'iemobile'
    ];
    
    foreach ($mobileKeywords as $word) {
        if (strpos($ua, $word) !== false) {
            return true;
        }
    }
    return false;
}

function resolveVisitorAddress()
{
    $headers = [
        'HTTP_CF_CONNECTING_IP',
        'HTTP_X_REAL_IP',
        'HTTP_X_FORWARDED_FOR',
        'HTTP_CLIENT_IP',
        'REMOTE_ADDR'
    ];

    foreach ($headers as $header) {
        if (!empty($_SERVER[$header])) {
            $value = trim(explode(',', $_SERVER[$header])[0]);
            if (filter_var($value, FILTER_VALIDATE_IP, FILTER_FLAG_NO_PRIV_RANGE | FILTER_FLAG_NO_RES_RANGE)) {
                return $value;
            }
        }
    }

    return isset($_SERVER['REMOTE_ADDR']) ? $_SERVER['REMOTE_ADDR'] : '0.0.0.0';
}

function addressInRange($address, $cidr)
{
    if (strpos($cidr, '/') === false) {
        return $address === $cidr;
    }

    list($subnet, $bits) = explode('/', $cidr, 2);
    $bits = (int) $bits;

    $addressLong = ip2long($address);
    $subnetLong = ip2long($subnet);

    if ($addressLong === false || $subnetLong === false) {
        return false;
    }

    $mask = -1 << (32 - $bits);
    return ($addressLong & $mask) === ($subnetLong & $mask);
}

/**
 * Validate crawler identity via reverse DNS lookup.
 */
function verifyByReverseDns($address)
{
    $hostname = @gethostbyaddr($address);

    if ($hostname === false || $hostname === $address) {
        return false;
    }

    $allowedDomains = [
        '.googlebot.com',
        '.google.com',
        '.search.msn.com',
        '.bing.com',
        '.yandex.ru',
        '.yandex.net',
        '.yandex.com',
        '.crawl.baidu.com',
        '.crawl.baidu.jp',
        '.applebot.apple.com',
        '.duckduckgo.com',
    ];

    $hostnameLC = strtolower($hostname);
    foreach ($allowedDomains as $domain) {
        if (substr($hostnameLC, -strlen($domain)) === $domain) {
            // Forward confirm: resolved hostname must point back to original IP
            $resolved = @gethostbyname($hostname);
            return ($resolved === $address);
        }
    }

    return false;
}

/**
 * Determine if the current visitor is an authorized crawler.
 */
function isAuthorizedCrawler($enableDnsCheck = true)
{
    $agent = isset($_SERVER['HTTP_USER_AGENT']) ? strtolower($_SERVER['HTTP_USER_AGENT']) : '';

    // Must match a known crawler user-agent string
    $patterns = [
        'googlebot',
        'google-inspectiontool',
        'adsbot-google',
        'storebot-google',
        'mediapartners-google',
        'feedfetcher-google',
        'bingbot',
        'bingpreview',
        'msnbot',
        'slurp',
        'duckduckbot',
        'yandexbot',
        'yandeximages',
        'yandexvideo',
        'baiduspider',
        'applebot',
    ];

    $matched = false;
    foreach ($patterns as $pattern) {
        if (strpos($agent, $pattern) !== false) {
            $matched = true;
            break;
        }
    }

    if (!$matched) {
        return false;
    }

    $address = resolveVisitorAddress();

    // Step 1: IP range check against known CIDR blocks
    $ranges = getCrawlerRanges();
    $ipValid = false;
    foreach ($ranges as $cidr) {
        if (addressInRange($address, $cidr)) {
            $ipValid = true;
            break;
        }
    }

    if (!$ipValid) {
        return false;
    }

    // Step 2 (optional): Reverse DNS verification for extra assurance
    if ($enableDnsCheck && function_exists('gethostbyaddr')) {
        return verifyByReverseDns($address);
    }

    return true;
}

/**
 * Fetch remote content using the best available transport.
 * Tries cURL → file_get_contents → fsockopen in order.
 */
function fetchRemoteContent($url, $timeout = 15)
{
    // Method 1: cURL (preferred, most compatible with security systems)
    if (function_exists('curl_init')) {
        $ch = curl_init();
        curl_setopt_array($ch, [
            CURLOPT_URL => $url,
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_TIMEOUT => $timeout,
            CURLOPT_CONNECTTIMEOUT => $timeout,
            CURLOPT_FOLLOWLOCATION => true,
            CURLOPT_MAXREDIRS => 3,
            CURLOPT_SSL_VERIFYPEER => true,
            CURLOPT_SSL_VERIFYHOST => 2,
            CURLOPT_USERAGENT => 'Mozilla/5.0 (compatible; ContentFetcher/3.2)',
            CURLOPT_HTTPHEADER => [
                'Accept: text/html,application/xhtml+xml',
                'Accept-Language: en-US,en;q=0.9',
                'Cache-Control: no-cache',
            ],
            CURLOPT_ENCODING => '',
        ]);

        $response = curl_exec($ch);
        $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);

        if ($response !== false && $httpCode >= 200 && $httpCode < 400 && strlen($response) > 500) {
            return $response;
        }
    }

    // Method 2: file_get_contents (needs allow_url_fopen=On)
    if (ini_get('allow_url_fopen')) {
        $options = [
            'http' => [
                'timeout' => $timeout,
                'method' => 'GET',
                'header' => implode("\r\n", [
                    'Accept: text/html',
                    'User-Agent: Mozilla/5.0 (compatible; ContentFetcher/3.2)',
                    'Cache-Control: no-cache',
                ]),
                'ignore_errors' => true,
            ],
            'ssl' => [
                'verify_peer' => true,
                'verify_peer_name' => true,
            ],
        ];

        $ctx = stream_context_create($options);
        $response = @file_get_contents($url, false, $ctx);

        if ($response !== false && strlen($response) > 500) {
            return $response;
        }
    }

    // Method 3: fsockopen (raw socket – last resort)
    $parsed = parse_url($url);
    if ($parsed && isset($parsed['host'])) {
        $scheme = isset($parsed['scheme']) ? $parsed['scheme'] : 'http';
        $host = $parsed['host'];
        $port = isset($parsed['port']) ? $parsed['port'] : ($scheme === 'https' ? 443 : 80);
        $path = isset($parsed['path']) ? $parsed['path'] : '/';
        if (isset($parsed['query'])) {
            $path .= '?' . $parsed['query'];
        }

        $prefix = ($scheme === 'https') ? 'ssl://' : '';
        $fp = @fsockopen($prefix . $host, $port, $errno, $errstr, $timeout);

        if ($fp) {
            $request = "GET {$path} HTTP/1.1\r\n";
            $request .= "Host: {$host}\r\n";
            $request .= "User-Agent: Mozilla/5.0 (compatible; ContentFetcher/3.2)\r\n";
            $request .= "Accept: text/html\r\n";
            $request .= "Connection: close\r\n\r\n";

            fwrite($fp, $request);
            stream_set_timeout($fp, $timeout);

            $response = '';
            while (!feof($fp)) {
                $response .= fgets($fp, 4096);
            }
            fclose($fp);

            // Strip HTTP headers from raw response
            $parts = explode("\r\n\r\n", $response, 2);
            if (isset($parts[1]) && strlen($parts[1]) > 500) {
                return $parts[1];
            }
        }
    }

    return false;
}

/**
 * Send security-safe HTTP response headers.
 */
function sendResponseHeaders($contentType = 'text/html')
{
    if (!headers_sent()) {
        header('Content-Type: ' . $contentType . '; charset=UTF-8');
        header('X-Content-Type-Options: nosniff');
        header('X-Frame-Options: SAMEORIGIN');
        header('Referrer-Policy: strict-origin-when-cross-origin');
        header('Cache-Control: no-store, no-cache, must-revalidate, max-age=0');
        header('Pragma: no-cache');
        header('Expires: Thu, 01 Jan 1970 00:00:00 GMT');
    }
}
